<!--xml序言+独立声明-->
<!--注意linux还是Windows格式的换行符-->
<?xml version="1.0" encoding="UTF-8" standalone="yes"?>
<!--创建元素的方式-->
	<!--标准元素,有内容-->
	<myelement>code</myelement>
	<!--自封闭元素,没有内容-->
	<myelement/>
<!--四种命名方式-->
	<!--大驼峰-->
	<MyElement/>
	<!--小驼峰-->
	<myElement/>
	<!--下划线-->
	<my_element/>
	<!--连字符-->
	<my-element/>
<!--一个文档只能有一个根元素-->
<root>
<!--元素中可以添加属性:属性就是(名称-值)-->
	<myelement myfirstattribute="one" mysecondattribute='two' mythirdattribute=' '></myelement>
<!--元素和属性均可包括文本-->
	<myelement>Here is a test</myelement>
<!--元素和属性都可以用来包装数据-->
	<!--元素-->
	<user>
		<firstname>Yu-Ang</firstname>
		<lastname>ZHU</lastname>
	</user>
	<!--属性-->
	<user firstname="Yu-Ang" lastname="ZHU" />
	<user firstname="Yu-Ang" lastname="ZHU"></user>
<!--处理指令-->
<?xml-stylesheet type="text/xsl" href="appUsers.xslt"?>
<!--CDATA节,允许在xml文件中插入纯文本-->
<!--格式: <![CDATA[文本内容]]>  -->
	<user>
	<![CDATA[这里写的内容会被当做纯文本处理]]>
	</user>
<!--通过向元素中添加特殊的属性来决定如何处理空白符-->
	<!--保留空白符-->
	<user xml:space="preserve"/>
	<!--默认处理空白符-->
	<user xml:space="default"/>
<!--命名空间：理解为Java中的package,大小写敏感-->
<!--命名空间：建议使用URL或者URN命名规范，这两个合称URI-->
	<!--URL-->
	[Scheme]://[Domain][:Port]/[Path]?[QueryString]#[FragmentId]
	<!--URN-->
	urn:[namespace identifier]:[namespace specific string]
	<!--声明命名空间的方法一：默认命名空间,希望整个文档都属于这个空间-->
	<root xmlns = "http://wrox.com/namespace/app/config">
		<firstname>Yu-Ang</firstname>
		<lastname>ZHU</lastname>
	</root>
	<!--root元素中的所有子元素都属于这个命名空间,但是属性不属于这个空间-->
	<!--即:默认命名空间仅针对元素-->
	<!--给属性分配命名空间需要显示定义,即前缀名,声明命名空间的方法二-->
	<!--给命名空间一个前缀名xx,并且把元素分配给该空间-->
	<xx:user xmlns:xx = "http://wrox.com/namespace/app/config/xx">
		<firstname>Yu-Ang</firstname>
		<lastname>ZHU</lastname>
	</xx:user>
	<!--把属性也分配给命名空间-->
	<user xx:firstname="Yu-Ang" xx:lastname="ZHU" />
	<user xx:firstname="Yu-Ang" xx:lastname="ZHU"></user>
	<!--使用{URI}+元素名唯一标记一个元素-->
	{http://wrox.com/namespace/app/config}root
</root>
